import { Card, CardGrid, Tabs, TabItem } from '@astrojs/starlight/components';

<CardGrid stagger>
	<Card title="Write normal code" icon="pencil">
        Without any need to do anything different:
        ```svelte
        <p>Hello world!</p>
        ```
	</Card>
	<Card title="Extract and translate" icon="translate">
        Optionally live translate using [Gemini](/guides/gemini)
        <Tabs>
            <TabItem label="es.po">
                ```diff lang="nginx" ins="¡Hola Mundo!"
                msgid "Hello world!"
                msgstr "¡Hola Mundo!"
                ```
            </TabItem>
            <TabItem label="en.po">
                ```nginx
                msgid "Hello world!"
                msgstr "Hello world!"
                ```
            </TabItem>
        </Tabs>
	</Card>
	<Card title="Compile catalogs" icon="rocket">
        Into arrays. No need for keys.
        <Tabs>
            <TabItem label="es.compiled.js">
                ```javascript
                export const data = ["¡Hola Mundo!"]
                ```
            </TabItem>
            <TabItem label="en.compiled.js">
                ```javascript
                export const data = ["Hello world!"]
                ```
            </TabItem>
        </Tabs>
	</Card>
	<Card title="Transform" icon="setting">
        Update your code to access the translations by index
        ```svelte
        <p>{_w_runtime_.t(0)}</p>
        ```
	</Card>
	<Card title="Render" icon="open-book">
        Load the desired compiled catalog, and hand it to your framework
        ```html
        <p>¡Hola Mundo!</p>
        ```
	</Card>
</CardGrid>
