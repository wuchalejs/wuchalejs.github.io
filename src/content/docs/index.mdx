import { Card, CardGrid, Tabs, TabItem, LinkButton } from '@astrojs/starlight/components';
import Supported from './landing-supported/index.astro'

<CardGrid stagger>
	<Card title="Write plain code" icon="pencil">
        Without any need to do anything different:
        ```svelte
        <p>Hello world!</p>
        ```
	</Card>
	<Card title="Extract and translate" icon="translate">
        Into PO files and optionally live translate using [AI](/guides/ai)
        <Tabs syncKey="locale">
            <TabItem label="English">
                ```po
                msgid "Hello world!"
                msgstr "Hello world!"
                ```
            </TabItem>
            <TabItem label="Spanish (translated)">
                ```diff lang="po" ins="¡Hola Mundo!"
                msgid "Hello world!"
                msgstr "¡Hola Mundo!"
                ```
            </TabItem>
        </Tabs>
	</Card>
	<Card title="Compile catalogs" icon="rocket">
        Into arrays. No need for keys.
        <Tabs syncKey="locale">
            <TabItem label="English">
                ```javascript
                export let c = ["Hello world!"]
                ```
            </TabItem>
            <TabItem label="Spanish (translated)">
                ```javascript
                export let c = ["¡Hola Mundo!"]
                ```
            </TabItem>
        </Tabs>
	</Card>
	<Card title="Transform" icon="setting">
        Update your code to access the translations by index
        ```svelte
        <p>{_w_runtime_.t(0)}</p>
        ```
	</Card>
	<Card title="Render" icon="open-book">
        Load the desired compiled catalog, and hand it to your framework
        ```html
        <p>¡Hola Mundo!</p>
        ```
	</Card>
</CardGrid>

<Card title="Inspiration">
    This project was inspired by [Lingui](https://lingui.dev/) especially some of
    its workflow. If you've used Lingui before, you'll find familiar concepts like
    extraction and compilation.

    Where `wuchale` differs, among other things, is that you don't need to change your
    code, catalogs compile smaller than any other tool (including Lingui's), and it
    integrates with a wider range of frameworks.
</Card>
